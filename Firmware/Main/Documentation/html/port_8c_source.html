203void vPortSetupRuntimeTimer() {
204 uint32_t period = (float)F_BUS_ACTUAL * (float)(100000/configTICK_RATE_HZ) * 0.0000005f;
205 uint32_t prescale = 0;
206 while (period > 32767) {
207 period = period >> 1;
208 if (++prescale > 7) {
209 prescale = 7; // when F_BUS is 150 MHz, longest
210 period = 32767; // period is 55922 us (~17.9 Hz)
211 break;
212 }
213 }
214 FLEXPWM2_FCTRL0 |= FLEXPWM_FCTRL0_FLVL(4); // logic high = fault
215 FLEXPWM2_FSTS0 = 0x0008; // clear fault status
216 FLEXPWM2_MCTRL |= FLEXPWM_MCTRL_CLDOK(4);
217 FLEXPWM2_SM2CTRL2 = FLEXPWM_SMCTRL2_INDEP;
218 FLEXPWM2_SM2CTRL = FLEXPWM_SMCTRL_HALF | FLEXPWM_SMCTRL_PRSC(prescale);
219 FLEXPWM2_SM2INIT = -period;
220 FLEXPWM2_SM2VAL0 = 0;
221 FLEXPWM2_SM2VAL1 = period;
222 FLEXPWM2_SM2VAL2 = 0;
223 FLEXPWM2_SM2VAL3 = 0;
224 FLEXPWM2_SM2VAL4 = 0;
225 FLEXPWM2_SM2VAL5 = 0;
226 FLEXPWM2_MCTRL |= FLEXPWM_MCTRL_LDOK(4);
227 FLEXPWM2_MCTRL &= ~FLEXPWM_MCTRL_RUN(4);
228 attachInterruptVector(IRQ_FLEXPWM2_2, &vPortRuntimeTimerCallback);
229 FLEXPWM2_SM2STS = FLEXPWM_SMSTS_RF;
230 FLEXPWM2_SM2INTEN = FLEXPWM_SMINTEN_RIE;
231 NVIC_ENABLE_IRQ(IRQ_FLEXPWM2_2);
232 FLEXPWM2_MCTRL |= FLEXPWM_MCTRL_RUN(4);
233}

242 */
243StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
244{
245 /* Simulate the stack frame as it would be created by a context switch
246 interrupt. */
247
248 /* Offset added to account for the way the MCU uses the stack on entry/exit
249 of interrupts, and to ensure alignment. */
250 pxTopOfStack -= 2;
251
252 *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
253 pxTopOfStack--;
254 *pxTopOfStack = ( ( StackType_t ) pxCode ) & portSTART_ADDRESS_MASK; /* PC */
255 pxTopOfStack--;
256 *pxTopOfStack = ( StackType_t ) portTASK_RETURN_ADDRESS; /* LR */
257
258 /* Save code space by skipping register initialisation. */
259 pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
260 *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
261
262 /* A save method is being used that requires each task to maintain its
263 own exec return value. */
264 pxTopOfStack--;
265 *pxTopOfStack = portINITIAL_EXC_RETURN;
266
267 pxTopOfStack -= 8; /* R11, R10, R9, R8, R7, R6, R5 and R4. */
268
269 return pxTopOfStack;
270}

278static void prvTaskExitError( void )
279{
280volatile uint32_t ulDummy = 0;
281
282 /* A function that implements a task must not exit or attempt to return to
283 its caller as there is nothing to return to. If a task wants to exit it
284 should instead call vTaskDelete( NULL ).
285
286 Artificially force an assert() to be triggered if configASSERT() is
287 defined, then stop here so application writers can catch the error. */
288 configASSERT( uxCriticalNesting == ~0UL );
289 portDISABLE_INTERRUPTS();
290 while( ulDummy == 0 )
291 {
292 /* This file calls prvTaskExitError() after the scheduler has been
293 started to remove a compiler warning about the function being defined
294 but never called. ulDummy is used purely to quieten other warnings
295 about code appearing after this function is called - making ulDummy
296 volatile makes the compiler think the function could return and
297 therefore not output an 'unreachable code' warning for code that appears
298 after it. */
299 }
300}

303void vPortSVCHandler( void )
304{
305 __asm volatile (
306 " ldr r3, pxCurrentTCBConst2 \n" /* Restore the context. */
307 " ldr r1, [r3] \n" /* Use pxCurrentTCBConst to get the pxCurrentTCB address. */
308 " ldr r0, [r1] \n" /* The first item in pxCurrentTCB is the task top of stack. */
309 " ldmia r0!, {r4-r11, r14} \n" /* Pop the registers that are not automatically saved on exception entry and the critical nesting count. */
310 " msr psp, r0 \n" /* Restore the task stack pointer. */
311 " isb \n"
312 " mov r0, #0 \n"
313 " msr basepri, r0 \n"
314 " bx r14 \n"
315 " \n"
316 " .align 4 \n"
317 "pxCurrentTCBConst2: .word pxCurrentTCB \n"
318 );
319}

322static void prvPortStartFirstTask( void )
323{
324 /* Start the first task. This also clears the bit that indicates the FPU is
325 in use in case the FPU was used before the scheduler was started - which
326 would otherwise result in the unnecessary leaving of space in the SVC stack
327 for lazy saving of FPU registers. */
328 __asm volatile(
329#if !INCLUDE_vTaskEndScheduler
330 " ldr r0, =0xE000ED08 \n" /* Use the NVIC offset register to locate the stack. */
331 " ldr r0, [r0] \n"
332 " ldr r0, [r0] \n"
333 " msr msp, r0 \n" /* Set the msp back to the start of the stack. */
334 " mov r0, #0 \n" /* Clear the bit that indicates the FPU is in use, see comment above. */
335 " msr control, r0 \n"
336#endif
337 " cpsie i \n" /* Globally enable interrupts. */
338 " cpsie f \n"
339 " dsb \n"
340 " isb \n"
341 " svc 0 \n" /* System call to start first task. */
342 " nop \n"
343 );
344}

347/*
348 * See header file for description.
349 */
350BaseType_t xPortStartScheduler( void )
351{
352 /* configMAX_SYSCALL_INTERRUPT_PRIORITY must not be set to 0.
353 See http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html */
354 configASSERT( configMAX_SYSCALL_INTERRUPT_PRIORITY );
355
356 /* This port can be used on all revisions of the Cortex-M7 core other than
357 the r0p1 parts. r0p1 parts should use the port from the
358 /source/portable/GCC/ARM_CM7/r0p1 directory. */
359 configASSERT( portCPUID != portCORTEX_M7_r0p1_ID );
360 configASSERT( portCPUID != portCORTEX_M7_r0p0_ID );
361
362 configASSERT( configTICK_RATE_HZ > 0 );
363
364 delay(100);
365
366 #if( configASSERT_DEFINED == 1 )
367 {
368 volatile uint32_t ulOriginalPriority;
369 volatile uint8_t * const pucFirstUserPriorityRegister = ( volatile uint8_t * const ) ( portNVIC_IP_REGISTERS_OFFSET_16 + portFIRST_USER_INTERRUPT_NUMBER );
370 volatile uint8_t ucMaxPriorityValue;
371
372 /* Determine the maximum priority from which ISR safe FreeRTOS API
373 functions can be called. ISR safe functions are those that end in
374 "FromISR". FreeRTOS maintains separate thread and ISR API functions to
375 ensure interrupt entry is as fast and simple as possible.
376
377 Save the interrupt priority value that is about to be clobbered. */
378 ulOriginalPriority = *pucFirstUserPriorityRegister;
379
380 /* Determine the number of priority bits available. First write to all
381 possible bits. */
382 *pucFirstUserPriorityRegister = portMAX_8_BIT_VALUE;
383
384 /* Read the value back to see how many bits stuck. */
385 ucMaxPriorityValue = *pucFirstUserPriorityRegister;
386
387 /* Use the same mask on the maximum system call priority. */
388 ucMaxSysCallPriority = configMAX_SYSCALL_INTERRUPT_PRIORITY & ucMaxPriorityValue;
389
390 /* Calculate the maximum acceptable priority group value for the number
391 of bits read back. */
392 ulMaxPRIGROUPValue = portMAX_PRIGROUP_BITS;
393 while( ( ucMaxPriorityValue & portTOP_BIT_OF_BYTE ) == portTOP_BIT_OF_BYTE )
394 {
395 ulMaxPRIGROUPValue--;
396 ucMaxPriorityValue <<= ( uint8_t ) 0x01;
397 }
398
399 #ifdef __NVIC_PRIO_BITS
400 {
401 /* Check the CMSIS configuration that defines the number of
402 priority bits matches the number of priority bits actually queried
403 from the hardware. */
404 configASSERT( ( portMAX_PRIGROUP_BITS - ulMaxPRIGROUPValue ) == __NVIC_PRIO_BITS );
405 }
406 #endif
407
408 #ifdef configPRIO_BITS
409 {
410 /* Check the FreeRTOS configuration that defines the number of
411 priority bits matches the number of priority bits actually queried
412 from the hardware. */
413 configASSERT( ( portMAX_PRIGROUP_BITS - ulMaxPRIGROUPValue ) == configPRIO_BITS );
414 }
415 #endif
416
417 /* Shift the priority group value back to its position within the AIRCR
418 register. */
419 ulMaxPRIGROUPValue <<= portPRIGROUP_SHIFT;
420 ulMaxPRIGROUPValue &= portPRIORITY_GROUP_MASK;
421
422 /* Restore the clobbered interrupt priority register to its original
423 value. */
424 *pucFirstUserPriorityRegister = ulOriginalPriority;
425 }
426 #endif /* conifgASSERT_DEFINED */
427
428 /* Make PendSV and SysTick the lowest priority interrupts. */
429 portNVIC_SYSPRI2_REG |= portNVIC_PENDSV_PRI;
430 portNVIC_SYSPRI2_REG |= portNVIC_SYSTICK_PRI;
431
432 /* Start the timer that generates the tick ISR. Interrupts are disabled
433 here already. */
434 vPortSetupTimerInterrupt();
435
436 /* Initialise the critical nesting count ready for the first task. */
437 uxCriticalNesting = 0;
438
439 vPortDisableVFP();
440
441 /* Lazy save always. */
442 *( portFPCCR ) |= portASPEN_AND_LSPEN_BITS;
443
444 vPortEnableVFP();
445
446#if INCLUDE_vTaskEndScheduler
447 if(setjmp(xJumpBuf) != 0 ) {
448 /* here we will get in case of call to vTaskEndScheduler() */
449 __asm volatile(
450 " movs r0, #1 \n" /* Switch back to the MSP stack. */
451 " msr CONTROL, r0 \n"
452 );
453 __asm volatile("dsb");
454 __asm volatile("isb");
455 return pdFALSE;
456 }
457#endif
458
459 /* Start the first task. */
460 prvPortStartFirstTask();
461
462 /* Should never get here as the tasks will now be executing! Call the task
463 exit error function to prevent compiler warnings about a static function
464 not being called in the case that the application writer overrides this
465 functionality by defining configTASK_RETURN_ADDRESS. Call
466 vTaskSwitchContext() so link time optimisation does not remove the
467 symbol. */
468 vTaskSwitchContext();
469 prvTaskExitError();
470
471 /* Should not get here! */
472 return 0;
473}

505void vPortEnterCritical( void )
506{
507 if(portFreeRTOSSchedulerStarted) portDISABLE_INTERRUPTS();
508 uxCriticalNesting++;
509
510 /* This is not the interrupt safe version of the enter critical function so
511 assert() if it is being called from an interrupt context. Only API
512 functions that end in "FromISR" can be used in an interrupt. Only assert if
513 the critical nesting count is 1 to protect against recursive calls if the
514 assert function also uses a critical section. */
515 if( uxCriticalNesting == 1 )
516 {
517 configASSERT( ( portNVIC_INT_CTRL_REG & portVECTACTIVE_MASK ) == 0 );
518 }
519}

533void xPortPendSVHandler( void )
534{
535 //pendablesrvreq_isr();
536 __asm volatile
537 (
538 " mrs r0, psp \n"
539 " isb \n"
540 " \n"
541 " ldr r3, pxCurrentTCBConst \n" /* Get the location of the current TCB. */
542 " ldr r2, [r3] \n"
543 " \n"
544 " tst r14, #0x10 \n" /* Is the task using the FPU context? If so, push high vfp registers. */
545 " it eq \n"
546 " vstmdbeq r0!, {s16-s31} \n"
547 " \n"
548 " stmdb r0!, {r4-r11, r14} \n" /* Save the core registers. */
549 " str r0, [r2] \n" /* Save the new top of stack into the first member of the TCB. */
550 " \n"
551 " stmdb sp!, {r0, r3} \n"
552 " mov r0, %0 \n"
553 " msr basepri, r0 \n"
554 " dsb \n"
555 " isb \n"
556 " bl vTaskSwitchContext \n"
557 " mov r0, #0 \n"
558 " msr basepri, r0 \n"
559 " ldmia sp!, {r0, r3} \n"
560 " \n"
561 " ldr r1, [r3] \n" /* The first item in pxCurrentTCB is the task top of stack. */
562 " ldr r0, [r1] \n"
563 " \n"
564 " ldmia r0!, {r4-r11, r14} \n" /* Pop the core registers. */
565 " \n"
566 " tst r14, #0x10 \n" /* Is the task using the FPU context? If so, pop the high vfp registers too. */
567 " it eq \n"
568 " vldmiaeq r0!, {s16-s31} \n"
569 " \n"
570 " msr psp, r0 \n"
571 " dsb \n"
572 " isb \n"
573 " \n"
574 #ifdef WORKAROUND_PMU_CM001 /* XMC4000 specific errata workaround. */
575 #if WORKAROUND_PMU_CM001 == 1
576 " push { r14 } \n"
577 " pop { pc } \n"
578 #endif
579 #endif
580 " \n"
581 " bx r14 \n"
582 " \n"
583 " .align 4 \n"
584 "pxCurrentTCBConst: .word pxCurrentTCB \n"
585 ::"i"(configMAX_SYSCALL_INTERRUPT_PRIORITY)
586 );
587}

590extern void systick_isr(void);
591void freertos_systick_isr(void)
592{
593 static uint8_t sub_tick_cnt = 0;
594 sub_tick_cnt++;
595
596 if(configTICK_RATE_HZ >= 1000) { // FreeRTOS is too fast
597 if(sub_tick_cnt >= configTICK_RATE_HZ/1000) {
598 systick_isr();
599 sub_tick_cnt = 0;
600 }
601 } else { // FreeRTOS is too slow
602 if(1000 % configTICK_RATE_HZ) { // Using the configTICK_RATE_HZ Hz systick
603 static uint32_t r = 0;
604 r += 1000 / configTICK_RATE_HZ;
605 for(uint32_t i=r ; i > 0 ; i--) {
606 systick_isr();
607 r -= 1;
608 }
609 } else { //In this case, we are using the 1kHz systick
610 systick_isr();
611 if(sub_tick_cnt < 1000/configTICK_RATE_HZ) {
612 return;
613 }
614 }
615
616 sub_tick_cnt = 0;
617 }
618
619 portSET_INTERRUPT_MASK_FROM_ISR();
620 {
621 /* Increment the RTOS tick. */
622 if( xTaskIncrementTick() != pdFALSE )
623 {
624 /* A context switch is required. Context switching is performed in
625 the PendSV interrupt. Pend the PendSV interrupt. */
626 portNVIC_INT_CTRL_REG = portNVIC_PENDSVSET_BIT;

637
638 __attribute__((weak)) void vPortSuppressTicksAndSleep( TickType_t xExpectedIdleTime )
639 {
640 uint32_t ulReloadValue, ulCompleteTickPeriods, ulCompletedSysTickDecrements;
641 TickType_t xModifiableIdleTime;
642
643 /* Make sure the SysTick reload value does not overflow the counter. */
644 if( xExpectedIdleTime > xMaximumPossibleSuppressedTicks )
645 {
646 xExpectedIdleTime = xMaximumPossibleSuppressedTicks;
647 }
648
649 /* Stop the SysTick momentarily. The time the SysTick is stopped for
650 is accounted for as best it can be, but using the tickless mode will
651 inevitably result in some tiny drift of the time maintained by the
652 kernel with respect to calendar time. */
653 portNVIC_SYSTICK_CTRL_REG &= ~portNVIC_SYSTICK_ENABLE_BIT;
654
655 /* Calculate the reload value required to wait xExpectedIdleTime
656 tick periods. -1 is used because this code will execute part way
657 through one of the tick periods. */
658 ulReloadValue = portNVIC_SYSTICK_CURRENT_VALUE_REG + ( ulTimerCountsForOneTick * ( xExpectedIdleTime - 1UL ) );
659 if( ulReloadValue > ulStoppedTimerCompensation )
660 {
661 ulReloadValue -= ulStoppedTimerCompensation;
662 }
663
664 /* Enter a critical section but don't use the taskENTER_CRITICAL()
665 method as that will mask interrupts that should exit sleep mode. */
666 __asm volatile( "cpsid i" ::: "memory" );
667 __asm volatile( "dsb" );
668 __asm volatile( "isb" );
669
670 /* If a context switch is pending or a task is waiting for the scheduler
671 to be unsuspended then abandon the low power entry. */
672 if( eTaskConfirmSleepModeStatus() == eAbortSleep )
673 {
674 /* Restart from whatever is left in the count register to complete
675 this tick period. */
676 portNVIC_SYSTICK_LOAD_REG = portNVIC_SYSTICK_CURRENT_VALUE_REG;
677
678 /* Restart SysTick. */
679 portNVIC_SYSTICK_CTRL_REG |= portNVIC_SYSTICK_ENABLE_BIT;
680
681 /* Reset the reload register to the value required for normal tick
682 periods. */
683 portNVIC_SYSTICK_LOAD_REG = ulTimerCountsForOneTick - 1UL;
684
685 /* Re-enable interrupts - see comments above the cpsid instruction()
686 above. */
687 __asm volatile( "cpsie i" ::: "memory" );
688 }
689 else
690 {
691 /* Set the new reload value. */
692 portNVIC_SYSTICK_LOAD_REG = ulReloadValue;
693
694 /* Clear the SysTick count flag and set the count value back to
695 zero. */
696 portNVIC_SYSTICK_CURRENT_VALUE_REG = 0UL;
697
698 /* Restart SysTick. */
699 portNVIC_SYSTICK_CTRL_REG |= portNVIC_SYSTICK_ENABLE_BIT;
700
701 /* Sleep until something happens. configPRE_SLEEP_PROCESSING() can
702 set its parameter to 0 to indicate that its implementation contains
703 its own wait for interrupt or wait for event instruction, and so wfi
704 should not be executed again. However, the original expected idle
705 time variable must remain unmodified, so a copy is taken. */
706 xModifiableIdleTime = xExpectedIdleTime;
707 configPRE_SLEEP_PROCESSING( xModifiableIdleTime );
708 if( xModifiableIdleTime > 0 )
709 {
710 __asm volatile( "dsb" ::: "memory" );
711 __asm volatile( "wfi" );
712 __asm volatile( "isb" );
713 }
714 configPOST_SLEEP_PROCESSING( xExpectedIdleTime );
715
716 /* Re-enable interrupts to allow the interrupt that brought the MCU
717 out of sleep mode to execute immediately. see comments above
718 __disable_interrupt() call above. */
719 __asm volatile( "cpsie i" ::: "memory" );
720 __asm volatile( "dsb" );
721 __asm volatile( "isb" );
722
723 /* Disable interrupts again because the clock is about to be stopped
724 and interrupts that execute while the clock is stopped will increase
725 any slippage between the time maintained by the RTOS and calendar
726 time. */
727 __asm volatile( "cpsid i" ::: "memory" );
728 __asm volatile( "dsb" );
729 __asm volatile( "isb" );
730
731 /* Disable the SysTick clock without reading the
732 portNVIC_SYSTICK_CTRL_REG register to ensure the
733 portNVIC_SYSTICK_COUNT_FLAG_BIT is not cleared if it is set. Again,
734 the time the SysTick is stopped for is accounted for as best it can
735 be, but using the tickless mode will inevitably result in some tiny
736 drift of the time maintained by the kernel with respect to calendar
737 time*/
738 portNVIC_SYSTICK_CTRL_REG = ( portNVIC_SYSTICK_CLK_BIT | portNVIC_SYSTICK_INT_BIT );
739
740 /* Determine if the SysTick clock has already counted to zero and
741 been set back to the current reload value (the reload back being
742 correct for the entire expected idle time) or if the SysTick is yet
743 to count to zero (in which case an interrupt other than the SysTick
744 must have brought the system out of sleep mode). */
745 if( ( portNVIC_SYSTICK_CTRL_REG & portNVIC_SYSTICK_COUNT_FLAG_BIT ) != 0 )
746 {
747 uint32_t ulCalculatedLoadValue;
748
749 /* The tick interrupt is already pending, and the SysTick count
750 reloaded with ulReloadValue. Reset the
751 portNVIC_SYSTICK_LOAD_REG with whatever remains of this tick
752 period. */
753 ulCalculatedLoadValue = ( ulTimerCountsForOneTick - 1UL ) - ( ulReloadValue - portNVIC_SYSTICK_CURRENT_VALUE_REG );
754
755 /* Don't allow a tiny value, or values that have somehow
756 underflowed because the post sleep hook did something
757 that took too long. */
758 if( ( ulCalculatedLoadValue < ulStoppedTimerCompensation ) || ( ulCalculatedLoadValue > ulTimerCountsForOneTick ) )
759 {
760 ulCalculatedLoadValue = ( ulTimerCountsForOneTick - 1UL );
761 }
762
763 portNVIC_SYSTICK_LOAD_REG = ulCalculatedLoadValue;
764
765 /* As the pending tick will be processed as soon as this
766 function exits, the tick value maintained by the tick is stepped
767 forward by one less than the time spent waiting. */
768 ulCompleteTickPeriods = xExpectedIdleTime - 1UL;
769 }
770 else
771 {
772 /* Something other than the tick interrupt ended the sleep.
773 Work out how long the sleep lasted rounded to complete tick
774 periods (not the ulReload value which accounted for part
775 ticks). */
776 ulCompletedSysTickDecrements = ( xExpectedIdleTime * ulTimerCountsForOneTick ) - portNVIC_SYSTICK_CURRENT_VALUE_REG;
777
778 /* How many complete tick periods passed while the processor
779 was waiting? */
780 ulCompleteTickPeriods = ulCompletedSysTickDecrements / ulTimerCountsForOneTick;
781
782 /* The reload value is set to whatever fraction of a single tick
783 period remains. */
784 portNVIC_SYSTICK_LOAD_REG = ( ( ulCompleteTickPeriods + 1UL ) * ulTimerCountsForOneTick ) - ulCompletedSysTickDecrements;
785 }
786
787 /* Restart SysTick so it runs from portNVIC_SYSTICK_LOAD_REG
788 again, then set portNVIC_SYSTICK_LOAD_REG back to its standard
789 value. */
790 portNVIC_SYSTICK_CURRENT_VALUE_REG = 0UL;
791 portNVIC_SYSTICK_CTRL_REG |= portNVIC_SYSTICK_ENABLE_BIT;
792 vTaskStepTick( ulCompleteTickPeriods );
793 portNVIC_SYSTICK_LOAD_REG = ulTimerCountsForOneTick - 1UL;
794
795 /* Exit with interrpts enabled. */
796 __asm volatile( "cpsie i" ::: "memory" );
797 }
798 }

860
